294void getFATChain(DWORD* dst, DWORD* difCh, CFHeader* cfh, BYTE* buf, DWORD secSize) {
295 for (int i = 0; i < 109; i++) {
296 if (cfh->DIFAT[i] == 0xffffffff) {
297 break;
298 }
299 dst[i] = cfh->DIFAT[i];
300 }
301 if (cfh->numOfFATSectors > 109) {
302 for (int i = 0; i < cfh->numOfDIFATSectors; i++) {
303 DWORD* difat = new DWORD[secSize / 4];
